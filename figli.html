from sklearn.linear_model import LinearRegression

# **Regressione solo per i maschi**
maschi = dataset[dataset['Gender'] == 0]  # Filtro corretto
X_maschi = maschi[['Father(cm)']]
y_maschi = maschi['Height(cm)']

# Creare e allenare il modello per i maschi
model_maschi = LinearRegression()
model_maschi.fit(X_maschi, y_maschi)
y_pred_maschi = model_maschi.predict(X_maschi)

# **Regressione solo per le femmine**
femmine = dataset[dataset['Gender'] == 1]  # Filtro corretto
X_femmine = femmine[['Father(cm)']]
y_femmine = femmine['Height(cm)']

# Creare e allenare il modello per le femmine
model_femmine = LinearRegression()
model_femmine.fit(X_femmine, y_femmine)
y_pred_femmine = model_femmine.predict(X_femmine)

# **Visualizzare i risultati**
plt.figure(figsize=(10, 6))

# Scatter e retta per i maschi
plt.scatter(X_maschi, y_maschi, color='blue', label='Maschi - Dati reali', alpha=0.5)
plt.plot(X_maschi, y_pred_maschi, color='darkblue', label='Maschi - Regressione')

# Scatter e retta per le femmine
plt.scatter(X_femmine, y_femmine, color='pink', label='Femmine - Dati reali', alpha=0.5)

Questa regressione visualizza la differenza di altezza tra i 2 generi

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression
from sklearn.model_selection import train_test_split

# Separiamo i dati per genere
figlio = dataset[dataset['Gender'] == 0]  # Figli maschi
figlia = dataset[dataset['Gender'] == 1]  # Figlie femmineed

X_maschi = figlio[['Father(cm)']].values  # Altezza del padre
y_maschi = figlio['Height(cm)'].values  # Altezza del figlio

X_femmina = figlia[['Mother(cm)']].values  # Altezza della madre
y_femmina = figlia['Height(cm)'].values  # Altezza della figlia

# Creiamo due modelli di regressione separati
model_m = LinearRegression()
model_f = LinearRegression()

# Alleniamo i modelli
model_m.fit(X_maschi, y_maschi)  # Modello per i maschi
model_f.fit(X_femmina, y_femmina)  # Modello per le femmine

# Fare predizioni
y_pred_maschi = model_m.predict(X_maschi)
y_pred_femmina = model_f.predict(X_femmina)

# Visualizzare i risultati
plt.figure(figsize=(10,6))
plt.scatter(X_maschi, y_maschi, color='blue', label='Dati figlio')  # Dati reali maschi
plt.plot(X_maschi, y_pred_maschi, color='red', label='Retta di regressione padre vs figlio')  # Regressione lineare maschi
plt.scatter(X_femmina, y_femmina, color='pink', label='Dati figlia')  # Dati reali femmine
plt.plot(X_femmina, y_pred_femmina, color='violet', label='Retta di regressione madre vs figlia')  # Regressione lineare femmine

plt.title("Regressione lineare: Statura dei genitori vs Statura dei figli")
plt.xlabel("Statura del genitore (in cm)")
plt.ylabel("Statura del figlio (in cm)")
plt.legend()
plt.show()
